   ```python
   # 逻辑回归预测
   import numpy as np
   
   def sigmoid(z):
       """Sigmoid激活函数"""
       return 1 / (1 + np.exp(-z))
   
   def logistic_regression(X, y, iterations=1000, alpha=0.01):
       m, n = X.shape
       theta = np.zeros((n, 1))
       
       for _ in range(iterations):
           # 前向传播
           z = np.dot(X, theta)
           h = sigmoid(z)
           
           # 计算梯度
           gradient = np.dot(X.T, (h - y)) / m
           
           # 更新参数
           theta = theta - alpha * gradient
       
       return theta
   ```

   **代码示例**：
   ```python
   def forward_propagation(X, W1, b1, W2, b2):
       """前向传播"""
       # 隐藏层计算
       Z1 = np.dot(W1, X) + b1
       A1 = np.tanh(Z1)  # 使用tanh激活
       
       # 输出层计算
       Z2 = np.dot(W2, A1) + b2
       A2 = sigmoid(Z2)
       
       cache = (X, W1, b1, Z1, A1, W2, b2, Z2)
       return A2, cache
   
   def backward_propagation(cache, Y):
       """反向传播"""
       X, W1, b1, Z1, A1, W2, b2, Z2 = cache
       m = X.shape[1]
       
       # 输出层梯度
       dZ2 = A2 - Y
       dW2 = np.dot(dZ2, A1.T) / m
       db2 = np.sum(dZ2, axis=1, keepdims=True) / m
       
       # 隐藏层梯度
       dA1 = np.dot(W2.T, dZ2)
       dZ1 = dA1 * (1 - A1**2)  # tanh的导数
       dW1 = np.dot(dZ1, X.T) / m
       db1 = np.sum(dZ1, axis=1, keepdims=True) / m
       
       return dW1, db1, dW2, db2
   ```

   ```python
   def dropout(A, keep_prob=0.8):
       """随机删除神经元"""
       # 生成掩码（保留概率为keep_prob）
       D = np.random.binomial(1, keep_prob, size=A.shape)
       
       # 应用掩码并缩放
       A_drop = A * D / keep_prob
       
       return A_drop, D
   ```

3. **代码示例**：
   ```python
   import openai
   
   def get_completion(prompt, model="gpt-3.5-turbo"):
       """调用OpenAI API"""
       messages = [{"role": "user", "content": prompt}]
       response = openai.ChatCompletion.create(
           model=model,
           messages=messages,
           temperature=0,  # 控制随机性
       )
       return response.choices[0].message["content"]
   
   # 使用示例：文本总结
   product_review = """
   购买了这个咖啡机已经一个月了...
   优点：快速加热，咖啡好喝
   缺点：噪音有点大
   """
   
   prompt = f"""
   用1句话总结以下评论：
   ```{product_review}```
   """
   
   summary = get_completion(prompt)
   print(summary)
   ```

## 常见问题解答

### Q1: 我需要很强的数学基础吗？

**A**: 不需要。虽然机器学习涉及数学，但吴恩达的课程强调直观理解而不是数学推导。需要的只是：
- 高中数学知识
- 基本的概率论概念
- 理解矩阵和向量（课程会教）

如果某些数学概念卡住了，可以：
1. 查看Khan Academy的相关视频
2. 临时跳过数学细节，继续学算法应用
3. 后续回过头来理解

### Q2: 应该先学哪门课程？

**A**: 取决于你的目标：

```
目标：理解AI基础
  ↓
选择：AI for Everyone

目标：系统学习机器学习，有编程基础
  ↓
选择：Machine Learning → Deep Learning Specialization

目标：快速上手大模型开发
  ↓
选择：Generative AI for Everyone → Prompt Engineering

目标：深入研究，数学基础好
  ↓
选择：CS229（Stanford研究生课程）
```

### Q3: Coursera上的课程可以免费学吗？

**A**: 可以。Coursera支持"免费旁听"（Audit），你可以：
- 看所有视频讲座
- 阅读所有课程资料
- 完成大部分作业

不能做的是：
- 获得正式证书
- 提交某些高级作业
- 参加同步课程

要获证书需要付费（通常$29-49/门课）。

### Q4: 用Python还是MATLAB？

**A**: **强烈推荐Python**原因：
1. 吴恩达新课程都是Python
2. 业界标准
3. 免费且生态完整
4. 学习资源丰富

老版本课程用MATLAB，但：
- 新版本已改为Python
- 即使看MATLAB版，概念相同，只需学Python语法

### Q5: 需要GPU吗？

**A**: 
- **Machine Learning课程**：不需要，CPU即可
- **Deep Learning课程的作业**：建议有GPU，但CPU也能完成（较慢）
- **实战项目**：根据数据量决定

**替代方案**：
- Google Colab（免费GPU）
- Kaggle Notebooks（免费GPU）
- AWS/Azure试用额度

### Q6: 多久能学完？

**A**: 取决于投入时间和基础：

```
有编程基础：
  Machine Learning课程      3-4个月 (每周10-15小时)
  Deep Learning专项         4-6个月 (每周10-15小时)
  总计：6-10个月

无编程基础：
  先学Python基础            1个月
  其他同上
  总计：7-11个月

只想快速了解：
  AI for Everyone + Generative AI  2周
```

### Q7: 学完后能做什么？

**A**: 

**能力水平**：
1. **Machine Learning完成** → 能独立做基础数据科学项目
2. **Deep Learning完成** → 能做图像/NLP项目
3. **大模型课程完成** → 能开发LLM应用

**就业方向**：
- 数据科学家
- 机器学习工程师
- AI产品经理

- AI创业者

**收入参考**（仅供参考）：

- 中级机器学习工程师：25K-40K/月
- 资深AI工程师：50K-100K+/月
