### QueryEngine

**文件：** `src/openharness/engine/query_engine.py`

QueryEngine 是整个 Agent 的核心引擎，负责管理对话历史和工具感知的模型循环。

```python
class QueryEngine:
    def __init__(self, *,
        api_client,           # API 客户端（支持流式）
        tool_registry,         # 工具注册表
        permission_checker,    # 权限检查器
        cwd,                  # 工作目录
        model,                # 模型名称
        system_prompt,         # 系统提示

        max_turns,            # 最大循环次数
        permission_prompt,    # 权限提示
        hook_executor,        # Hook 执行器
    ):
```

**核心方法：**

1. `submit_message(prompt)` → `AsyncIterator[StreamEvent]`
   - 添加用户消息到历史
   - 启动 QueryContext 执行循环
   - **流式产出事件**

2. `continue_pending()`
   - 恢复中断的工具循环
   - 不添加新消息

3. `has_pending_continuation()`
   - 检查是否有待续的工具结果

**run_query 流程（推测）：**
```python
async def run_query(context: QueryContext, messages: list[ConversationMessage]):
    while True:
        # 1. 构建 API 请求
        response = await api_client.stream(messages, tools)

        # 2. 处理停止原因
        if response.stop_reason == "tool_use":
            # 3. 执行工具
            for tool_call in response.tool_uses:
                # 权限检查 → Hook → 执行 → 结果
                result = await execute_tool(tool_call)
                messages.append(ToolResultBlock(...))
        else:
            break  # 模型完成

        # 4. 检查 max_turns
        if turns >= max_turns:
            break
```

class BaseTool(ABC):
    name: str                    # 工具名称
    description: str              # 工具描述
    input_model: type[BaseModel] # Pydantic 输入模型

    @abstractmethod
    async def execute(
        self,
        arguments: BaseModel,
        context: ToolExecutionContext
    ) -> ToolResult:
        """执行工具"""

    def is_read_only(self, arguments: BaseModel) -> bool:
        """判断是否只读"""
        return False

    def to_api_schema(self) -> dict:
        """转换为 Anthropic API schema"""
        return {
            "name": self.name,
            "description": self.description,
            "input_schema": self.input_model.model_json_schema(),
        }

def add_memory_entry(cwd: str | Path, title: str, content: str) -> Path:
    """
    1. 在 memory/ 目录创建 {slug}.md
    2. 在 MEMORY.md 中添加索引链接
    """
    memory_dir = get_project_memory_dir(cwd)
    slug = sub(r"[^a-zA-Z0-9]+", "_", title.strip().lower())
    path = memory_dir / f"{slug}.md"
    path.write_text(content.strip() + "\n")

    # 更新 MEMORY.md 索引
    entrypoint = get_memory_entrypoint(cwd)
    existing = entrypoint.read_text() if entrypoint.exists() else "# Memory Index\n"
    if path.name not in existing:
        existing += f"\n- [{title}]({path.name})\n"
        entrypoint.write_text(existing)

    return path

### PermissionChecker.evaluate()

```python
def evaluate(self, tool_name, *, is_read_only, file_path, command) -> PermissionDecision:
    # 1. 检查显式工具拒绝列表
    if tool_name in self._settings.denied_tools:
        return PermissionDecision(allowed=False, reason=f"{tool_name} is explicitly denied")

    # 2. 检查显式工具允许列表
    if tool_name in self._settings.allowed_tools:
        return PermissionDecision(allowed=True)

    # 3. 检查路径规则 (glob 匹配)
    if file_path and self._path_rules:
        for rule in self._path_rules:
            if fnmatch.fnmatch(file_path, rule.pattern):
                if not rule.allow:
                    return PermissionDecision(allowed=False)

    # 4. 检查命令拒绝模式
    if command:
        for pattern in self._settings.denied_commands:
            if fnmatch.fnmatch(command, pattern):
                return PermissionDecision(allowed=False)

    # 5. 根据权限模式决定
    if mode == FULL_AUTO:
        return PermissionDecision(allowed=True)

    if is_read_only:
        return PermissionDecision(allowed=True)

    if mode == PLAN:
        return PermissionDecision(allowed=False, reason="Plan mode blocks mutating tools")

    # DEFAULT: 需要确认
    return PermissionDecision(
        allowed=False,
        requires_confirmation=True,
        reason="Mutating tools require user confirmation"
    )
```
