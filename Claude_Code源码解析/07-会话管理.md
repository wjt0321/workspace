## 7.4 历史读取

### 逆序读取（最新在前）

```typescript
async function* makeLogEntryReader(): AsyncGenerator<LogEntry> {
  // 1. 先读取内存中的待写入条目
  for (let i = pendingEntries.length - 1; i >= 0; i--) {
    yield pendingEntries[i]!
  }
  
  // 2. 再从磁盘文件逆序读取
  const historyPath = join(getClaudeConfigHomeDir(), 'history.jsonl')
  
  for await (const line of readLinesReverse(historyPath)) {
    const entry = jsonParse(line)
    yield entry
  }
}
```

## 7.5 粘贴内容处理

### 粘贴引用

```typescript
// 用户粘贴大段文本时，用引用代替
"[Pasted text #1 +50 lines]"

// 图片粘贴
"[Image #2]"
```

### 引用解析

```typescript
function parseReferences(input: string): Reference[] {
  const pattern = /\[(Pasted text|Image) #(\d+)(?: \+\d+ lines)?\]/g
  const matches = [...input.matchAll(pattern)]
  return matches.map(m => ({
    id: parseInt(m[2]),
    type: m[1],  // 'Pasted text' | 'Image'
    match: m[0]
  }))
}
```

### 大文件存储

```typescript
// 大于 1KB 的粘贴内容单独存储
const MAX_INLINE_LENGTH = 1024

if (content.length > MAX_INLINE_LENGTH) {
  const hash = hashPastedText(content)

  return { id, contentHash: hash }
} else {
  return { id, content }
}
```

```typescript
const MAX_HISTORY_ITEMS = 100  // 最多保留 100 条

async function pruneHistory(): Promise<void> {
  const entries = await readAllHistory()
  
  if (entries.length > MAX_HISTORY_ITEMS) {
    const toDelete = entries.slice(0, entries.length - MAX_HISTORY_ITEMS)
```

```typescript
async function exportSession(sessionId: string): Promise<string> {
  const messages = await readTranscript(sessionId)
  
  return messages.map(m => {
    if (m.type === 'user') return `**User**: ${m.content}`
    if (m.type === 'assistant') return `**Assistant**: ${m.content}`
    if (m.type === 'tool_result') return `**Tool**: ${m.tool}\n\`\`\`\n${m.result}\n\`\`\``
  }).join('\n\n')
```
