### buildTool 工厂函数

```typescript
async function buildTool(name: string, definition: ToolDefinition): Promise<Tool> {
  return {
    name,
    description: definition.description,
    input_schema: definition.input_schema,
    output_schema: definition.output_schema,
    // 核心：把 JSON Schema 转换成验证函数
    validate: createValidator(definition.input_schema),
    // 执行函数
    execute: definition.execute
  }
}
```

## 2.5 权限系统

### 权限检查接口

```typescript
type CanUseToolFn = (
  toolName: string,
  toolInput: object,
  context: ToolPermissionContext
) => Promise<PermissionResult>

type PermissionResult = 
  | { result: 'ask' }           // 需要询问用户
  | { result: 'allow' }          // 允许
  | { result: 'deny' }           // 拒绝
  | { result: 'bypass' }         // 绕过（管理员）
```

### 权限模式

```typescript
type PermissionMode = 
  | 'default'      // 默认，每次询问
  | 'auto'         // 自动，根据规则决定
  | 'bypass'       // 绕过，所有允许
  | 'dangerous'    // 危险模式
```

### 重试机制

```typescript
async function executeWithRetry(
  tool: Tool,
  input: object,
  maxRetries: number = 3
): Promise<ToolResult> {
  for (let i = 0; i < maxRetries; i++) {
    try {
      return await tool.execute(input)
    } catch (error) {
      if (isRetryable(error) && i < maxRetries - 1) {
        await delay(1000 * (i + 1))
        continue
      }
      throw error
    }
  }
}
```
